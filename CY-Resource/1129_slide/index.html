SQL Commands

Create Table

CREATE TABLE table_name(parameters);

Parameters

  • column name + datatype
  • datatype: INTEGER, TEXT...

Example

CREATE TABLE member(name TEXT, email TEXT, phone TEXT, pass TEXT);

Insert Row To a Table

ALTER TABLE table_name ADD COLUMN parameters;

Example

ALTER TABLE member ADD COLUMN facebook TEXT;

Insert Data To a Table

INSERT INTO table_name(parameters) VALUES (...);

Example

INSERT INTO member(name,email,phone,pass)
VALUES ('Jason', 'chunyou0830@gmail.com', '0953110830', 'chunyou0830');
INSERT INTO member(name,email,phone,pass)
VALUES ('Jimmy', 'cba22265886abc@yahoo.com.tw', '0986285154', '12345');
INSERT INTO member(name,email,phone,pass)
VALUES ('', 'empty@uhh.what', '', '');

Get Data From a Table

SELECT column_name FROM db_name;

Example

SELECT * FROM member;
SELECT name, email FROM member;

Conditions

WHERE

SELECT column_name FROM db_name WHERE condition;
SELECT name, email FROM member WHERE name = 'Jason';

LIKE

SELECT name, email FROM member WHERE name LIKE 'J_';

Much more conditions

  • AND
  • OR
  • BETWEEN ... AND ...

Update Data From a Table

UPDATE db_name SET col_name = ... WHERE index_col_name = ...;

Example

UPDATE member SET name = 'Jason Yang' WHERE email = 'chunyou0830@gmail.com';

Delete Data From a Table

DELETE FROM db_name WHERE condition;

Example

DELETE FROM member WHERE email = 'empty@uhh.what';

PHP Functions

Connect with host

mysql_connect(HOST, USERNAME, PASSWORD);

Example

mysql_connect("localhost", "root", "");

Select DB

mysql_select_db(DB_NAME);

Example

mysql_select_db("test");

Query

mysql_query(QUERY);

Example

mysql_query("SELECT * FROM member");

Fetch Array

mysql_fetch_array(QUERY_RETURN);

Example

$result = mysql_query("SELECT * FROM member");
$row = mysql_fetch_array($result);

Counting Rows

mysql_num_rows(QUERY_RETURN);

Example

$result = mysql_query("SELECT * FROM member");
$row_nums = mysql_num_rows($result);

Supplement - MD5

Just a hashing function.

md5(STRING);

Example

$result = md5("myPassword");
